基类希望其派生类进行覆盖的函数，定义为虚函数

#include <iostream>
using namespace std;

class Animal{
public:
  void sound(){
    cout << "??" << endl;
  }
};



class Cat : public Animal{
public:
  void sound(){
    cout << "Miao Miao" << endl;
  }
};

class Dog : public Animal{
public:
  void sound(){
    cout << "Wang Wang" << endl;
  }
};

int main(int argc, char **argv){
  Animal *animal;
  animal = new Cat;
  animal->sound();
  animal = new Dog;
  animal->sound();
}

??

可见animal指针虽然指向子类Cat和Dog，但是并没用他们的sound方法。基类使用virtual关键字

void tree_preorder_re(TreeNode *node, Tree *tree, void (*visit)(Tree *, TreeNode *))
{
    if (!node) return;
    visit(tree, node);
    tree_preorder_re(node->left, tree, visit);
    tree_preorder_re(node->right, tree, visit);
}

中序和后序，不过是调整一下visit的位置，如果不用递归的写法，改用栈来实现，其实没有那么直观。 首先按照函数调用的关系，入栈的套路都是一样的。

#include <iostream>
#include <memory>
#include <iterator>

using namespace std;

class Foo{
    public:
    Foo() {p = new int[10];cout << "new@" << p <<endl;}
    ~Foo() {cout << "release@" << p << endl;delete [] p;p=nullptr;}
    int *p;
};

void smartptr()
{
    shared_ptr<Foo> sp1 = make_shared<Foo>();
    auto sp2 = make_shared<Foo>();
}
void ptr()
{
    auto p = new Foo;
}
int main(int argc,char* argv[])
{
    smartptr();
    ptr();
